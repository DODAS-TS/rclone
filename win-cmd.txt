rclone:
	go build -v --ldflags "-s -X github.com/rclone/rclone/fs.Version=$(TAG)" $(BUILDTAGS) $(BUILD_ARGS)
	mkdir -p ("{0}\bin" -f (go env GOPATH))
	cp ("rclone{0}" -f (go env GOEXE)) ("{0}\bin\rclone{1}.new" -f (go env GOPATH, GOEXE))
	mv -v ("{0}\bin\rclone{1}.new" -f (go env GOPATH, GOEXE)) ("{0}\bin\rclone{1}" -f (go env GOPATH, GOEXE))

test_all:
